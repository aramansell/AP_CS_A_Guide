Tic-Tac-Toe — Full Walkthrough

This is the program your teacher builds live. It uses a 2D array as the game board, with methods to place marks, check for winners, and display the board. Use this page as a reference.

The Goal

Build a two-player Tic-Tac-Toe game with a 3x3 board. Let players take turns placing X and O. Check for a winner after every move.

  0 1 2
0  | | 
 ---+---+---
1  | | 
 ---+---+---
2  | | 

Player X, enter row and col: 1 1
...
X wins!

Step 1: The Board (2D Array)

public class TicTacToe {
    private char[][] board;
    private char currentPlayer;

    public TicTacToe() {
        board = new char[3][3];
        // Initialize with empty spaces
        for (int r = 0; r < 3; r++) {
            for (int c = 0; c < 3; c++) {
                board[r][c] = ' ';
            }
        }
        currentPlayer = 'X';
    }

Why char and not String? Each cell holds one character. char is simpler, lighter, and == comparison works (no .equals() needed). X, O, or space — three possible values.

Step 2: Display the Board

    public void display() {
        System.out.println("  0 1 2");
        for (int r = 0; r < 3; r++) {
            System.out.print(r + " ");
            for (int c = 0; c < 3; c++) {
                System.out.print(board[r][c]);
                if (c < 2) System.out.print("|");
            }
            System.out.println();
            if (r < 2) System.out.println(" ---+---+---");
        }
    }

Each cell prints the char. Bars between columns. Dashed line between rows. The row/col labels help players choose coordinates.

Step 3: Place a Mark

    public boolean placeMark(int row, int col) {
        if (row < 0 || row > 2 || col < 0 || col > 2) {
            System.out.println("Out of bounds! Use 0-2.");
            return false;
        }
        if (board[row][col] != ' ') {
            System.out.println("Cell is already taken!");
            return false;
        }
        board[row][col] = currentPlayer;
        return true;
    }

Step 4: Check for Winner

Three ways to win: rows, columns, diagonals. Check all eight lines.

    public boolean checkWin() {
        // Check rows
        for (int r = 0; r < 3; r++) {
            if (board[r][0] != ' ' &&
                board[r][0] == board[r][1] &&
                board[r][1] == board[r][2])
                return true;
        }
        // Check columns
        for (int c = 0; c < 3; c++) {
            if (board[0][c] != ' ' &&
                board[0][c] == board[1][c] &&
                board[1][c] == board[2][c])
                return true;
        }
        // Check diagonals
        if (board[0][0] != ' ' &&
            board[0][0] == board[1][1] &&
            board[1][1] == board[2][2])
            return true;
        if (board[0][2] != ' ' &&
            board[0][2] == board[1][1] &&
            board[1][1] == board[2][0])
            return true;

        return false;
    }

First check != ' '. An empty row (three spaces) would technically "match" but shouldn't count as a win. Skip empty cells.

Step 5: Game Loop

    public void play() {
        Scanner input = new Scanner(System.in);
        int moves = 0;

        while (moves < 9) {
            display();
            System.out.println("Player " + currentPlayer
                               + ", enter row and col: ");
            int row = input.nextInt();
            int col = input.nextInt();

            if (placeMark(row, col)) {
                moves++;
                if (checkWin()) {
                    display();
                    System.out.println(currentPlayer + " wins!");
                    return;
                }
                // Switch player
                currentPlayer = (currentPlayer == 'X') ? 'O' : 'X';
            }
        }
        display();
        System.out.println("It's a tie!");
    }

The Full Program

import java.util.Scanner;

public class TicTacToe {
    private char[][] board;
    private char currentPlayer;

    public TicTacToe() {
        board = new char[3][3];
        for (int r = 0; r < 3; r++)
            for (int c = 0; c < 3; c++)
                board[r][c] = ' ';
        currentPlayer = 'X';
    }

    public void display() {
        System.out.println("  0 1 2");
        for (int r = 0; r < 3; r++) {
            System.out.print(r + " ");
            for (int c = 0; c < 3; c++) {
                System.out.print(board[r][c]);
                if (c < 2) System.out.print("|");
            }
            System.out.println();
            if (r < 2) System.out.println(" ---+---+---");
        }
    }

    public boolean placeMark(int row, int col) {
        if (row < 0 || row > 2 || col < 0 || col > 2) return false;
        if (board[row][col] != ' ') return false;
        board[row][col] = currentPlayer;
        return true;
    }

    public boolean checkWin() {
        for (int r = 0; r < 3; r++)
            if (board[r][0] != ' ' && board[r][0] == board[r][1]
                && board[r][1] == board[r][2]) return true;
        for (int c = 0; c < 3; c++)
            if (board[0][c] != ' ' && board[0][c] == board[1][c]
                && board[1][c] == board[2][c]) return true;
        if (board[0][0] != ' ' && board[0][0] == board[1][1]
            && board[1][1] == board[2][2]) return true;
        if (board[0][2] != ' ' && board[0][2] == board[1][1]
            && board[1][1] == board[2][0]) return true;
        return false;
    }

    public void play() {
        Scanner input = new Scanner(System.in);
        int moves = 0;
        while (moves < 9) {
            display();
            System.out.print("Player " + currentPlayer + ": row col? ");
            int r = input.nextInt(), c = input.nextInt();
            if (placeMark(r, c)) {
                moves++;
                if (checkWin()) { display(); System.out.println(currentPlayer + " wins!"); return; }
                currentPlayer = (currentPlayer == 'X') ? 'O' : 'X';
            }
        }
        System.out.println("Tie!");
    }

    public static void main(String[] args) {
        new TicTacToe().play();
    }
}
